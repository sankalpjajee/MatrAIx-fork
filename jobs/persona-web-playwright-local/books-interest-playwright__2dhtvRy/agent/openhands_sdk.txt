import json
import asyncio
from playwright.async_api import async_playwright

async def browse():
    async with async_playwright() as p:
        browser = await p.chromium.launch(headless=True)
        page = await browser.new_page()

        await page.goto("https://books.toscrape.com/")
        await page.wait_for_load_state("networkidle")

        # Get page title to confirm site loaded
        title = await page.title()
        print(f"Page title: {title}")

        # Explore category list
        categories = await page.query_selector_all("ul.nav-list > li > ul > li >
a")
        cat_names = []
        for c in categories:
            name = (await c.inner_text()).strip()
            cat_names.append(name)
        print("Categories:", cat_names[:20])

        # Look for 'Business' or 'Self Help' or 'Travel' categories
        target_cats = ["Business", "Self Help", "Travel", "Psychology", 
"Nonfiction", "Parenting"]
        chosen_cat_link = None
        chosen_cat_name = None
        for c in categories:
            name = (await c.inner_text()).strip()
            if name in target_cats:
                chosen_cat_link = c
                chosen_cat_name = name
                break

        if chosen_cat_link:
            print(f"Clicking category: {chosen_cat_name}")
            await chosen_cat_link.click()
            await page.wait_for_load_state("networkidle")
        else:
            print("No target category found, browsing homepage")

        # Get books on the page
        books = await page.query_selector_all("article.product_pod")
        print(f"Found {len(books)} books")

        book_list = []
        for book in books:
            title_el = await book.query_selector("h3 > a")
            price_el = await book.query_selector("p.price_color")
            if title_el and price_el:
                # The title attribute holds the full title
                t = await title_el.get_attribute("title")
                if not t:
                    t = (await title_el.inner_text()).strip()
                price = (await price_el.inner_text()).strip()
                book_list.append({"title": t, "price": price})
                print(f"  {t} — {price}")

        await browser.close()
        return book_list

async def browse():
    async with async_playwright() as p:
        browser = await p.chromium.launch(headless=True)
        page = await browser.new_page()

        await page.goto("https://books.toscrape.com/")
        await page.wait_for_load_state("networkidle")

        # Get page title to confirm site loaded
        title = await page.title()
        print(f"Page title: {title}")

        # Explore category list
        categories = await page.query_selector_all("ul.nav-list > li > ul > li >
a")
        cat_names = []
        for c in categories:
            name = (await c.inner_text()).strip()
            cat_names.append(name)
        print("Categories:", cat_names[:20])

        # Look for 'Business' or 'Self Help' or 'Travel' categories
        target_cats = ["Business", "Self Help", "Travel", "Psychology", 
"Nonfiction", "Parenting"]
        chosen_cat_link = None
        chosen_cat_name = None
        for c in categories:
            name = (await c.inner_text()).strip()
            if name in target_cats:
                chosen_cat_link = c
                chosen_cat_name = name
                break

        if chosen_cat_link:
            print(f"Clicking category: {chosen_cat_name}")
            await chosen_cat_link.click()
            await page.wait_for_load_state("networkidle")
        else:
            print("No target category found, browsing homepage")

        # Get books on the page
        books = await page.query_selector_all("article.product_pod")
        print(f"Found {len(books)} books")

        book_list = []
        for book in books:
            title_el = await book.query_selector("h3 > a")
            price_el = await book.query_selector("p.price_color")
            if title_el and price_el:
                # The title attribute holds the full title
                t = await title_el.get_attribute("title")
                if not t:
                    t = (await title_el.inner_text()).strip()
                price = (await price_el.inner_text()).strip()
                book_list.append({"title": t, "price": price})
                print(f"  {t} — {price}")

        await browser.close()
        return book_list
